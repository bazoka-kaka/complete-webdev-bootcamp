- challenge:
add in our housekeeper constructor a method called 
"clean" which is gonna alert "cleaning in progress"

solution:
function HouseKeeper(yearsOfExp, name, cleaningRepetoire) {
    this.yearsOfExp = yearsOfExp;
    this.name = name;
    this.cleaningRepetoire = cleaningRepetoire;
    this.clean = function() {
        alert("Cleaning in Progress");
    }
}

let houseKeeper1 = new HouseKeeper(12, "Jane", ["Bathroom", "Bedroom"]);

console.log(houseKeeper1);
HouseKeeper.clean();
