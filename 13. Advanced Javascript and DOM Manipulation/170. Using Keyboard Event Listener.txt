USING KEYBOARD EVENT LISTENERS TO CHECK FOR KEY PRESSES 
=======================================================

- note:
TL;DR keypress is now deprecated, you should 
use keydown instead.

- creating a keypressed listener:
document.addEventListener("keydown", function (event) {
  console.log(event);
});

note:
this is gonna output the event and we can see everything 
that is triggered in there

- final code:
let drum = document.querySelectorAll(".drum");

function checkButton(button) {
  let audio;
  switch (button) {
    case "w":
      audio = new Audio("sounds/crash.mp3");
      break;
    case "a":
      audio = new Audio("sounds/kick-bass.mp3");
      break;
    case "s":
      audio = new Audio("sounds/snare.mp3");
      break;
    case "d":
      audio = new Audio("sounds/tom-1.mp3");
      break;
    case "j":
      audio = new Audio("sounds/tom-2.mp3");
      break;
    case "k":
      audio = new Audio("sounds/tom-3.mp3");
      break;
    case "l":
      audio = new Audio("sounds/tom-4.mp3");
      break;
    default:
  }
  audio.play();
}

for (let i = 0; i < drum.length; ++i) {
  drum[i].addEventListener("click", function () {
    checkButton(this.innerHTML);
  });
}

document.addEventListener("keydown", function (event) {
  checkButton(event.key);
});
